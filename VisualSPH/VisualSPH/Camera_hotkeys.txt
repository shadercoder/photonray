Camera HotKeys:

//-------------------------------------------------------------------------------
//							Moves:
//-------------------------------------------------------------------------------

'W' -   forward
'S' -   back
'A' -	left
'D' -	right

//-------------------------------------------------------------------------------
//							Rotation:
//-------------------------------------------------------------------------------

('<-' / '->') 			    -  	Rotate around Up vector;

('Keyboard_up' / 'Keyboard_down')   -	Rotate around right vector(Slope);

('N' / 'M')                         -	Rotate around look vector;     

//-------------------------------------------------------------------------------
//							Screenshots:
//-------------------------------------------------------------------------------

'P' - Take a screenshot
